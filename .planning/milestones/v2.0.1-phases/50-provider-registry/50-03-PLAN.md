    4. Implement `HandleTestConnection()`:
       ```csharp
       private async Task HandleTestConnection()
       {
           if (EditTarget == null || _testingConnection) return;

           _testingConnection = true;
           _testSuccess = null;
           _testErrorMessage = null;
           StateHasChanged();

           _testCts = new CancellationTokenSource(TimeSpan.FromSeconds(30));
           try
           {
               var result = await ProviderRegistry.TestConnectionAsync(EditTarget.Slug, _testCts.Token);
               _testSuccess = result.Success;
               _testErrorMessage = result.ErrorMessage;
           }
           catch (OperationCanceledException)
           {
               _testSuccess = false;
               _testErrorMessage = "Request timed out";
           }
           catch (Exception ex)
           {
               _testSuccess = false;
               _testErrorMessage = ex.Message;
           }
           finally
           {
               _testingConnection = false;
               _testCts?.Dispose();
               _testCts = null;
               StateHasChanged();
           }

           // Auto-clear success after 5 seconds
           if (_testSuccess == true)
           {
               _ = Task.Run(async () =>
               {
                   await Task.Delay(5000);
                   await InvokeAsync(() =>
                   {
                       if (_testSuccess == true)
                       {
                           _testSuccess = null;
                           StateHasChanged();
                       }
                   });
               });
           }
       }
       ```
