```csharp
namespace OpenAnima.Core.Memory;

public class MemoryRecallService : IMemoryRecallService
{
    private const int MaxContentCharsPerNode = 500;
    private const int MaxTotalChars = 6000;

    private readonly IMemoryGraph _memoryGraph;
    private readonly ILogger<MemoryRecallService> _logger;

    public MemoryRecallService(IMemoryGraph memoryGraph, ILogger<MemoryRecallService> logger) { ... }

    public async Task<RecalledMemoryResult> RecallAsync(string animaId, string context, CancellationToken ct = default)
    {
        // Implementation steps (all in this method):
        // 1. Get disclosure nodes: await _memoryGraph.GetDisclosureNodesAsync(animaId, ct)
        // 2. Match disclosure: DisclosureMatcher.Match(disclosureNodes, context)
        // 3. Rebuild glossary: await _memoryGraph.RebuildGlossaryAsync(animaId, ct)
        // 4. Find glossary matches: _memoryGraph.FindGlossaryMatches(animaId, context)
        // 5. For glossary matches, need to load actual nodes -- use GetNodeAsync for each unique URI
        // 6. Deduplicate by URI using Dictionary<string, RecalledNode>:
        //    - If URI exists from disclosure and glossary also matches, merge reason: "disclosure + glossary: {keyword}"
        //    - Disclosure entries: reason = "disclosure", RecallType = "Disclosure"
        //    - Glossary entries: reason = "glossary: {keyword}", RecallType = "Glossary"
        // 7. Priority sort: Boot > Disclosure > Glossary (use RecallType for ordering)
        //    - Priority map: "Boot" = 0, "Disclosure" = 1, "Glossary" = 2
        //    - Within same type, sort by node.UpdatedAt descending (string comparison works for ISO 8601)
        // 8. Truncate each node content: TruncatedContent = node.Content.Length > 500 ? node.Content[..500] : node.Content
        // 9. Apply total character budget: iterate sorted nodes, accumulate TruncatedContent.Length, stop when sum exceeds 6000
        // 10. Return RecalledMemoryResult with filtered Nodes list
    }
}
```

**Key implementation details:**
- Glossary `FindGlossaryMatches` returns `(string Keyword, string Uri)` tuples. To get the full MemoryNode, call `_memoryGraph.GetNodeAsync(animaId, uri, ct)` for each unique URI. Skip nulls.
- URI deduplication dictionary: key = node.Uri, value = RecalledNode. When merging, concatenate reasons.
- Boot nodes are NOT loaded in this service -- they are handled separately by BootMemoryInjector in RunService. The RecallType "Boot" exists for future use but is not populated by RecallAsync. Priority sort still works because disclosure and glossary nodes never have RecallType "Boot".
- Log at Debug level: "Recalled {Count} nodes for Anima {AnimaId} ({DisclosureCount} disclosure, {GlossaryCount} glossary)"
