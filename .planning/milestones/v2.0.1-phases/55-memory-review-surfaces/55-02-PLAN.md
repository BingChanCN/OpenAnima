```csharp
// ── Snapshot History ──

private async Task ToggleSnapshotHistory()
{
    _snapshotHistoryExpanded = !_snapshotHistoryExpanded;
    if (_snapshotHistoryExpanded && !_snapshotsLoaded)
    {
        _snapshots = await MemoryGraphService.GetSnapshotsAsync(Node.AnimaId, Node.Uri);
        _snapshotsLoaded = true;
    }
}

// ── Provenance ──

private async Task ToggleStepDetails()
{
    _stepDetailsExpanded = !_stepDetailsExpanded;
    if (_stepDetailsExpanded && _stepDetails == null && Node.SourceStepId != null)
    {
        try
        {
            _stepDetails = await RunRepository.GetStepByIdAsync(Node.SourceStepId);
            if (_stepDetails == null) _stepFetchError = true;
        }
        catch
        {
            _stepFetchError = true;
        }
    }
}

// ── Relationships ──

private async Task ToggleRelationships()
{
    _relationshipsExpanded = !_relationshipsExpanded;
    if (_relationshipsExpanded && !_edgesLoaded)
    {
        try
        {
            _outgoingEdges = await MemoryGraphService.GetEdgesAsync(Node.AnimaId, Node.Uri);
            _incomingEdges = await MemoryGraphService.GetIncomingEdgesAsync(Node.AnimaId, Node.Uri);
            _edgesLoaded = true;

            // Pre-fetch counterpart node content for tooltips
            var counterpartUris = _outgoingEdges.Select(e => e.ToUri)
                .Concat(_incomingEdges.Select(e => e.FromUri))
                .Distinct()
                .Where(uri => uri != Node.Uri);

            var fetchTasks = counterpartUris.Select(async uri =>
            {
                var node = await MemoryGraphService.GetNodeAsync(Node.AnimaId, uri);
                if (node != null) _nodeContentCache[uri] = node.Content;
            });
            await Task.WhenAll(fetchTasks);
        }
        catch
        {
            _edgeFetchError = true;
        }
    }
}

private async Task NavigateToNode(string uri)
{
    await OnNavigateToUri.InvokeAsync(uri);

// ── Restore ──

private void RequestRestore(MemorySnapshot snapshot)
{
    _pendingRestoreSnapshot = snapshot;
    _showRestoreConfirm = true;
}

private void CancelRestore()
{
    _showRestoreConfirm = false;
    _pendingRestoreSnapshot = null;
}

private async Task ExecuteRestore()
{
    if (_pendingRestoreSnapshot == null) return;

    var restored = Node with
    {
        Content = _pendingRestoreSnapshot.Content,
        UpdatedAt = DateTimeOffset.UtcNow.ToString("O")
    };
    await OnSaveNode.InvokeAsync(restored);

    _showRestoreConfirm = false;
    _pendingRestoreSnapshot = null;

    // Reset snapshots so they reload with the new state
    _snapshotsLoaded = false;
    if (_snapshotHistoryExpanded)
    {
        _snapshots = await MemoryGraphService.GetSnapshotsAsync(Node.AnimaId, Node.Uri);
        _snapshotsLoaded = true;
    }

private async Task HideRestoreFlash()
{
    await Task.Delay(1500);
    _showSaveFlash = false;
    await InvokeAsync(StateHasChanged);
}

private static string Truncate(string? text, int maxLength)
{
    if (string.IsNullOrEmpty(text)) return "";
    return text.Length <= maxLength ? text : text[..maxLength] + "...";
}
```
