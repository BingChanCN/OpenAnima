<objective>
Add a "Living Memory" configuration section to the Settings page that lets users select which LLM provider and model powers living memory sedimentation. The section writes `sedimentProviderSlug` and `sedimentModelId` to `AnimaModuleConfigService("Sedimentation")` scoped per active Anima, which immediately activates the dormant `SedimentationService.CallProductionLlmAsync` pipeline.

Purpose: Without this UI, the sedimentation pipeline silently skips on every call because the two config keys are never populated. This is the only missing piece to activate LIVM-01 end-to-end.

Output: Modified Settings.razor with new section, CSS additions, i18n keys in both locales, and an integration test proving config round-trip activates sedimentation.
</objective>

    ```csharp
    private void LoadSedimentConfig()
    {
        if (string.IsNullOrEmpty(AnimaContext.ActiveAnimaId)) return;
        var config = ConfigService.GetConfig(AnimaContext.ActiveAnimaId, "Sedimentation");
        _sedimentProviderSlug = config.GetValueOrDefault("sedimentProviderSlug", "");
        _sedimentModelId = config.GetValueOrDefault("sedimentModelId", "");

        // Check if saved provider is now disabled
        _sedimentProviderDisabled = false;
        if (!string.IsNullOrEmpty(_sedimentProviderSlug))
        {
            var provider = LlmRegistry.GetProvider(_sedimentProviderSlug);
            _sedimentProviderDisabled = provider == null || !provider.IsEnabled;
        }
    }
    ```

    Add `OnActiveAnimaChanged` handler:

    Add dropdown change handlers:
    ```csharp
    private async Task HandleSedimentProviderChanged(ChangeEventArgs e)
    {
        _sedimentProviderSlug = e.Value?.ToString() ?? "";
        _sedimentModelId = ""; // cascade reset
        _sedimentProviderDisabled = false;
        await SaveSedimentConfigAsync();
    }

    private async Task HandleSedimentModelChanged(ChangeEventArgs e)
    {
        _sedimentModelId = e.Value?.ToString() ?? "";
        await SaveSedimentConfigAsync();
    }

    private async Task SaveSedimentConfigAsync()
    {
        if (string.IsNullOrEmpty(AnimaContext.ActiveAnimaId)) return;
        await ConfigService.SetConfigAsync(
            AnimaContext.ActiveAnimaId,
            "Sedimentation",
            new Dictionary<string, string>
            {
                ["sedimentProviderSlug"] = _sedimentProviderSlug,
                ["sedimentModelId"] = _sedimentModelId
            });
        await ShowSavedIndicator();
    }

    private async Task ShowSavedIndicator()
    {
        _savedIndicatorCts?.Cancel();
        _savedIndicatorCts = new CancellationTokenSource();
        _sedimentSaved = true;
        StateHasChanged();
        try
        {
            await Task.Delay(2000, _savedIndicatorCts.Token);
            _sedimentSaved = false;
            await InvokeAsync(StateHasChanged);
        }
        catch (TaskCanceledException) { }
    }
    ```

    ```csharp
    public void Dispose()

        AnimaContext.ActiveAnimaChanged -= OnActiveAnimaChanged;
    ```
