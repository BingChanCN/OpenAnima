    **Step 2 — Create `ConnectionContextMenu.razor`:**

    Create `src/OpenAnima.Core/Components/Shared/ConnectionContextMenu.razor` following the exact pattern of `ModuleContextMenu.razor`:

    ```razor
    @inject IStringLocalizer<SharedResources> L
    @inject LanguageService LangSvc
    @implements IDisposable

    @if (IsVisible)
    {
        <div class="context-menu-backdrop" @onclick="HandleClose"></div>
        <div class="context-menu" style="left: @(X)px; top: @(Y)px;" @onclick:stopPropagation="true">
            <button class="context-menu-item context-menu-item--danger" @onclick="HandleDelete">@L["Editor.Connection.Delete"]</button>
        </div>
    }

    @code {
        [Parameter] public bool IsVisible { get; set; }
        [Parameter] public double X { get; set; }
        [Parameter] public double Y { get; set; }
        [Parameter] public EventCallback OnDelete { get; set; }
        [Parameter] public EventCallback OnClose { get; set; }

        protected override void OnInitialized()
        {
            LangSvc.LanguageChanged += OnLanguageChanged;
        }

        private void OnLanguageChanged() => InvokeAsync(StateHasChanged);

        private async Task HandleDelete()
        {
            await OnClose.InvokeAsync();
            await OnDelete.InvokeAsync();
        }

        private async Task HandleClose()
        {
            await OnClose.InvokeAsync();
        }

        public void Dispose()
        {
            LangSvc.LanguageChanged -= OnLanguageChanged;
        }
    }
    ```

    Add a handler method:
    ```csharp
    private async Task HandleContextMenu(MouseEventArgs e)
    {

        }
    }
    ```

    Update the invisible hit-test `<path>` element (the first path, the one with `stroke="transparent"` and `stroke-width="12"`) to add the context menu handler:
    ```razor
    <path d="@_pathData"
          stroke="transparent"
          stroke-width="12"
          fill="none"
          pointer-events="@(IsPreview ? "none" : "stroke")"
          @onclick="HandleClick"
          @oncontextmenu="HandleContextMenu"
          @oncontextmenu:preventDefault
          class="@(IsPreview ? "" : "cursor-pointer")" />
    ```

    Note: `@oncontextmenu:preventDefault` prevents the browser's native context menu from appearing on the hit-test path.

    5b. Add a handler method for connection right-click:
    ```csharp
    private void HandleConnectionContextMenu(PortConnection connection, MouseEventArgs e)
    {
        _contextMenuConnection = connection;
        _connectionContextMenuX = e.ClientX;
        _connectionContextMenuY = e.ClientY;
        _showConnectionContextMenu = true;
        StateHasChanged();
    }
    ```

    5c. Add handler methods for context menu actions:
    ```csharp
    private void HandleConnectionContextMenuDelete()
    {
        if (_contextMenuConnection != null)
        {
            _state.RemoveConnection(
                _contextMenuConnection.SourceModuleId,
                _contextMenuConnection.SourcePortName,
                _contextMenuConnection.TargetModuleId,
                _contextMenuConnection.TargetPortName);
        }
        _showConnectionContextMenu = false;
        _contextMenuConnection = null;
    }

    private void HandleConnectionContextMenuClose()
    {
        _showConnectionContextMenu = false;
        _contextMenuConnection = null;
    }
    ```
