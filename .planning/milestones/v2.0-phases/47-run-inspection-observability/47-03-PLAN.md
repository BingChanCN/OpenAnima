    NEW STATE FIELDS (add to existing @code block):
    ```csharp

    ```

    NEW COMPUTED PROPERTIES:
    ```csharp
    private IReadOnlyList<string> _moduleNames => _steps.Select(s => s.ModuleName).Distinct().OrderBy(n => n).ToList();
    private IReadOnlyList<string> _statuses => _steps.Select(s => s.Status).Distinct().OrderBy(s => s).ToList();
    private IReadOnlyList<string> _chainIds => _steps.Where(s => !string.IsNullOrEmpty(s.PropagationId)).Select(s => s.PropagationId).Distinct().OrderBy(c => c).ToList();
    private bool _hasActiveFilters => _filterModule != null || _filterStatus != null || _filterChain != null;

    private List<TimelineEntry> _filteredTimeline => _timeline.Where(e =>
    {
        if (e.IsStateEvent) return _filterModule == null && _filterStatus == null && _filterChain == null;
        var step = e.Step!;
        if (_filterModule != null && step.ModuleName != _filterModule) return false;
        if (_filterStatus != null && step.Status != _filterStatus) return false;
        if (_filterChain != null && step.PropagationId != _filterChain) return false;
        return true;
    }).ToList();
    ```

    NEW EVENT HANDLERS:
    ```csharp
    private void HandleModuleFilter(string? value) { _filterModule = value; }
    private void HandleStatusFilter(string? value) { _filterStatus = value; }
    private void HandleChainFilter(string? value) { _filterChain = value; _activeChainId = null; }
    private void HandleClearFilters() { _filterModule = null; _filterStatus = null; _filterChain = null; _activeChainId = null; }
    private void HandleChainClick(string? propagationId)
    {
        if (string.IsNullOrEmpty(propagationId)) return;
        _activeChainId = _activeChainId == propagationId ? null : propagationId;
    }
    private void HandleChainFilterShortcut(string propagationId)
    {
        _filterChain = propagationId;
        _activeChainId = null;
    }
    ```
