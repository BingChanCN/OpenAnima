           [Parameter] public MemoryNode Node { get; set; } = default!;
           [Parameter] public EventCallback<MemoryNode> OnSaveNode { get; set; }
           [Parameter] public EventCallback<string> OnDeleteNode { get; set; }

           private string EditContent { get; set; } = "";
           private string? EditTrigger { get; set; }
           private List<string> _keywords = new();
           private string _newKeyword = "";
           private bool _showSaveFlash;

           protected override void OnParametersSet()
           {
               EditContent = Node.Content;
               EditTrigger = Node.DisclosureTrigger;
               _keywords = ParseKeywords(Node.Keywords);
           }

           private static List<string> ParseKeywords(string? json)
           {
               if (string.IsNullOrEmpty(json)) return new();
               try { return System.Text.Json.JsonSerializer.Deserialize<List<string>>(json) ?? new(); }
               catch { return new(); }
           }

           private static string SerializeKeywords(List<string> keywords)
               => System.Text.Json.JsonSerializer.Serialize(keywords);

           private void RemoveKeyword(string kw)
           {
               _keywords.Remove(kw);
           }

           private void HandleKeywordKeyDown(KeyboardEventArgs e)
           {
               if ((e.Key == "Enter" || e.Key == ",") && !string.IsNullOrWhiteSpace(_newKeyword))
               {
                   var trimmed = _newKeyword.Trim().TrimEnd(',');
                   if (!string.IsNullOrEmpty(trimmed) && !_keywords.Contains(trimmed))
                       _keywords.Add(trimmed);
                   _newKeyword = "";
               }
           }

           private async Task OnSave()
           {
               var updated = Node with
               {
                   Content = EditContent,
                   DisclosureTrigger = string.IsNullOrWhiteSpace(EditTrigger) ? null : EditTrigger,
                   Keywords = _keywords.Count > 0 ? SerializeKeywords(_keywords) : null,
                   UpdatedAt = DateTimeOffset.UtcNow.ToString("O")
               };
               await OnSaveNode.InvokeAsync(updated);
               _showSaveFlash = true;
               _ = HideSaveFlash();
           }

           private async Task HideSaveFlash()
           {
               await Task.Delay(1500);
               _showSaveFlash = false;
               await InvokeAsync(StateHasChanged);
           }

           private async Task OnDeleteClick()
           {
               await OnDeleteNode.InvokeAsync(Node.Uri);
           }

           private static string FormatTimestamp(string iso)
           {
               if (DateTimeOffset.TryParse(iso, out var dt))
                   return dt.LocalDateTime.ToString("yyyy-MM-dd HH:mm:ss");
               return iso;
           }
       }

       @using OpenAnima.Core.Memory
       @using OpenAnima.Core.Anima

                      @bind="_searchFilter"
                      @bind:event="oninput" />
               @if (!string.IsNullOrEmpty(_searchFilter))
               {
                   <button class="search-clear" @onclick="ClearSearch" aria-label="Clear search">×</button>

                           </div>

               </div>

               <div class="node-detail-panel">

                   <div class="confirm-actions">

       @code {
           private List<MemoryNode> _treeNodes = new();
           private string? _selectedUri;
           private MemoryNode? _selectedNode;
           private string _searchFilter = "";
           private bool _isCreatingNew;
           private string _newUri = "";
           private string _newContent = "";
           private bool _showDeleteConfirm;
           private string? _pendingDeleteUri;
           private ElementReference _uriInputRef;

           private IEnumerable<MemoryNode> FilteredNodes =>
               string.IsNullOrEmpty(_searchFilter)
                   ? _treeNodes
                   : _treeNodes.Where(n => n.Uri.Contains(_searchFilter, StringComparison.OrdinalIgnoreCase));

           protected override async Task OnInitializedAsync()
           {
               await LoadNodes();
           }

           private async Task LoadNodes()
           {
               var animaId = AnimaContext.ActiveAnimaId;
               if (animaId == null) return;
               _treeNodes = (await MemoryGraphService.GetAllNodesAsync(animaId)).OrderBy(n => n.Uri).ToList();
           }

           private async Task SelectNode(string uri)
           {
               _selectedUri = uri;
               _isCreatingNew = false;
               var animaId = AnimaContext.ActiveAnimaId;
               if (animaId != null)
                   _selectedNode = await MemoryGraphService.GetNodeAsync(animaId, uri);
           }

           private void StartNewNode()
           {
               _isCreatingNew = true;
               _selectedNode = null;
               _selectedUri = null;
               _newUri = "";
               _newContent = "";
           }

           private void CancelNewNode()
           {
               _isCreatingNew = false;
           }

           private async Task CreateNewNode()
           {
               var animaId = AnimaContext.ActiveAnimaId;
               if (animaId == null || string.IsNullOrWhiteSpace(_newUri)) return;
               var now = DateTimeOffset.UtcNow.ToString("O");
               var node = new MemoryNode
               {
                   Uri = _newUri.Trim(),
                   AnimaId = animaId,
                   Content = _newContent,
                   CreatedAt = now,
                   UpdatedAt = now
               };
               await MemoryGraphService.WriteNodeAsync(node);
               _isCreatingNew = false;
               await LoadNodes();
               await SelectNode(node.Uri);
           }

           private async Task SaveNode(MemoryNode updated)
           {
               await MemoryGraphService.WriteNodeAsync(updated);
               await LoadNodes();
               _selectedNode = updated;
           }

           private void ConfirmDelete(string uri)
           {
               _pendingDeleteUri = uri;
               _showDeleteConfirm = true;
           }

           private void CancelDelete()
           {
               _showDeleteConfirm = false;
               _pendingDeleteUri = null;
           }

           private async Task ExecuteDelete()
           {
               var animaId = AnimaContext.ActiveAnimaId;
               if (animaId != null && _pendingDeleteUri != null)
               {
                   await MemoryGraphService.DeleteNodeAsync(animaId, _pendingDeleteUri);
                   _selectedNode = null;
                   _selectedUri = null;
                   await LoadNodes();
               }
               _showDeleteConfirm = false;
               _pendingDeleteUri = null;
           }

           private void ClearSearch()
           {
               _searchFilter = "";
           }
       }
