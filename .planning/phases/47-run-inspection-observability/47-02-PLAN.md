    MERGE TIMELINE METHOD:
    ```csharp
    private void RebuildTimeline()
    {
        var entries = new List<TimelineEntry>();
        foreach (var s in _steps)
            entries.Add(new TimelineEntry(s.OccurredAt, false, s, null));
        foreach (var e in _stateEvents)
            entries.Add(new TimelineEntry(e.OccurredAt, true, null, e));
        _timeline = entries.OrderBy(e => e.OccurredAt).ThenBy(e => e.IsStateEvent ? 0 : 1).ToList();
    }
    ```

    ON INITIALIZED:
    ```csharp
    protected override async Task OnInitializedAsync()
    {
        _run = await RunService.GetRunByIdAsync(RunId);
        if (_run == null) { _notFound = true; return; }

        _steps = (await Repository.GetStepsByRunIdAsync(RunId)).ToList();
        _stateEvents = (await Repository.GetStateEventsByRunIdAsync(RunId)).ToList();
        RebuildTimeline();

        _hubConnection = new HubConnectionBuilder()
            .WithUrl(Navigation.ToAbsoluteUri("/hubs/runtime"))
            .WithAutomaticReconnect()
            .Build();

        _hubConnection.On<string, string, string, string, string, int?>("ReceiveStepCompleted",
            async (animaId, runId, stepId, moduleName, status, durationMs) =>
            {
                if (runId != RunId) return;
                var updatedSteps = await Repository.GetStepsByRunIdAsync(RunId);
                _steps = updatedSteps.ToList();
                RebuildTimeline();
                await InvokeAsync(StateHasChanged);
            });

        _hubConnection.On<string, string, string, string?>("ReceiveRunStateChanged",
            async (animaId, runId, state, reason) =>
            {
                if (runId != RunId) return;
                var parsed = Enum.Parse<RunState>(state);
                _run = _run! with { CurrentState = parsed };
                var updatedEvents = await Repository.GetStateEventsByRunIdAsync(RunId);
                _stateEvents = updatedEvents.ToList();
                RebuildTimeline();
                await InvokeAsync(StateHasChanged);
            });

        await _hubConnection.StartAsync();
    }
    ```

    TOGGLE METHODS:
    ```csharp
    private void ToggleStep(string stepId)
    {
        _expandedStepId = _expandedStepId == stepId ? null : stepId;
    }

    private void HighlightChain(string? propagationId)
    {
        if (string.IsNullOrEmpty(propagationId)) return;
        _activeChainId = _activeChainId == propagationId ? null : propagationId;
    }
    ```

    DISPOSE:
    ```csharp
    public async ValueTask DisposeAsync()

    }
    ```
