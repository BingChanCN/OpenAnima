```csharp
        return _availableModules.Where(m =>
            m.Name.Contains(_searchFilter, StringComparison.OrdinalIgnoreCase) ||
```

```csharp
    var key = $"Module.DisplayName.{moduleName}";
    var localized = L[key];
    // If no translation found, IStringLocalizer returns the key itself — fall back to class name
    return localized.ResourceNotFound ? moduleName : localized.Value;
}
```
