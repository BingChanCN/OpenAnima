       - `WriteNodeAsync(MemoryNode node, CancellationToken ct)`:
         ```csharp
         await using var conn = _factory.CreateConnection();
         await conn.OpenAsync(ct);
         // Check if exists
         var existing = await conn.QueryFirstOrDefaultAsync<MemoryNode>(
             "SELECT uri AS Uri, anima_id AS AnimaId, content AS Content, disclosure_trigger AS DisclosureTrigger, keywords AS Keywords, source_artifact_id AS SourceArtifactId, source_step_id AS SourceStepId, created_at AS CreatedAt, updated_at AS UpdatedAt FROM memory_nodes WHERE uri = @Uri AND anima_id = @AnimaId",
             new { node.Uri, node.AnimaId });
         if (existing != null)
         {
             // Snapshot old content
             await conn.ExecuteAsync(
                 "INSERT INTO memory_snapshots (uri, anima_id, content, snapshot_at) VALUES (@Uri, @AnimaId, @Content, @SnapshotAt)",
                 new { existing.Uri, existing.AnimaId, existing.Content, SnapshotAt = DateTimeOffset.UtcNow.ToString("O") });
             // Prune to last 10
             await conn.ExecuteAsync(
                 "DELETE FROM memory_snapshots WHERE uri = @Uri AND anima_id = @AnimaId AND id NOT IN (SELECT id FROM memory_snapshots WHERE uri = @Uri AND anima_id = @AnimaId ORDER BY id DESC LIMIT 10)",
                 new { node.Uri, node.AnimaId });
             // Update
             await conn.ExecuteAsync(
                 "UPDATE memory_nodes SET content = @Content, disclosure_trigger = @DisclosureTrigger, keywords = @Keywords, source_artifact_id = @SourceArtifactId, source_step_id = @SourceStepId, updated_at = @UpdatedAt WHERE uri = @Uri AND anima_id = @AnimaId",
                 new { node.Content, node.DisclosureTrigger, node.Keywords, node.SourceArtifactId, node.SourceStepId, node.UpdatedAt, node.Uri, node.AnimaId });

         var nodes = await GetAllNodesAsync(animaId, ct);

         foreach (var n in nodes)
         {
             if (string.IsNullOrEmpty(n.Keywords)) continue;

             if (keywords == null) continue;
             foreach (var kw in keywords)

         ```
