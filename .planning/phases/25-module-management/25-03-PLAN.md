**Event handlers (add):**
```csharp
private void OpenContextMenu(MouseEventArgs e, PluginRegistryEntry module)
{
    _contextModuleName = module.Manifest.Name;
    _contextModuleEnabled = AnimaContext.ActiveAnimaId != null
        && ModuleStateService.IsModuleEnabled(AnimaContext.ActiveAnimaId, module.Manifest.Name);
    _contextMenuX = e.ClientX;
    _contextMenuY = e.ClientY;
    _showContextMenu = true;
}

private async Task HandleEnable()
{
    _showContextMenu = false;
    if (AnimaContext.ActiveAnimaId == null || _contextModuleName == null) return;
    await ModuleStateService.SetModuleEnabled(AnimaContext.ActiveAnimaId, _contextModuleName, true);
    StateHasChanged();
}

private async Task HandleDisable()
{
    _showContextMenu = false;
    if (AnimaContext.ActiveAnimaId == null || _contextModuleName == null) return;
    await ModuleStateService.SetModuleEnabled(AnimaContext.ActiveAnimaId, _contextModuleName, false);
    StateHasChanged();
}

private void ShowSidebar(PluginRegistryEntry module)
{
    _sidebarModule = module;
    _showSidebar = true;
}

private void CloseSidebar()
{
    _showSidebar = false;
}

private async Task HandleSidebarEnableDisable()
{
    if (AnimaContext.ActiveAnimaId == null || _sidebarModule == null) return;
    var isEnabled = ModuleStateService.IsModuleEnabled(AnimaContext.ActiveAnimaId, _sidebarModule.Manifest.Name);
    await ModuleStateService.SetModuleEnabled(AnimaContext.ActiveAnimaId, _sidebarModule.Manifest.Name, !isEnabled);
    StateHasChanged();
}
```

**Subscribe to ActiveAnimaChanged (OnInitialized):**
```csharp
AnimaContext.ActiveAnimaChanged += HandleActiveAnimaChanged;

private void HandleActiveAnimaChanged()
{
    InvokeAsync(StateHasChanged);
}
```

**Unsubscribe in DisposeAsync:**
```csharp
AnimaContext.ActiveAnimaChanged -= HandleActiveAnimaChanged;
```

**Add components at bottom:**
```razor
&lt;ModuleContextMenu IsVisible="_showContextMenu"
                  X="_contextMenuX"
                  Y="_contextMenuY"
                  ModuleName="@(_contextModuleName ?? "")"
                  IsEnabled="_contextModuleEnabled"
                  OnEnable="HandleEnable"
                  OnDisable="HandleDisable"
                  OnUninstall="() =&gt; UnloadModule(_contextModuleName)"
                  OnViewDetails="() =&gt; { _showSidebar = true; _sidebarModule = ModuleService.GetAllModules().FirstOrDefault(m =&gt; m.Manifest.Name == _contextModuleName); _showContextMenu = false; }"
                  OnClose="() =&gt; _showContextMenu = false" /&gt;

&lt;ModuleDetailSidebar IsVisible="_showSidebar"
                     SelectedModule="_sidebarModule"
                     OnClose="CloseSidebar"
                     OnEnableDisable="HandleSidebarEnableDisable"
                     OnUninstall="() =&gt; UnloadModule(_sidebarModule?.Manifest.Name)" /&gt;
```
