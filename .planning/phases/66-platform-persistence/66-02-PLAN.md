# Plan 66-02: Viewport persistence (pan/zoom/scale per Anima)

## Objective

Persist the wiring editor viewport (pan X, pan Y, scale) to a JSON file `{animaId}.viewport.json` in the same directory as `{name}.json` wiring config files. Restore the viewport when the application starts. Viewport saves use a 1000ms debounce independent from the 500ms config auto-save debounce.

## must_haves

- On `UpdatePan()` and `UpdateScale()` calls in `EditorStateService`, a 1000ms debounced save writes `{animaId}.viewport.json` to the config directory
- The viewport file format is: `{"scale":1.0,"panX":0.0,"panY":0.0}`
- On startup, `WiringInitializationService.StartAsync` loads and applies the saved viewport for the active Anima
- Per-Anima: different Animas have separate viewport files named by their ID
- `IViewportPersistenceService` has `SaveAsync(animaId, scale, panX, panY)` and `LoadAsync(animaId)` returning `(double Scale, double PanX, double PanY)?`

```csharp
namespace OpenAnima.Core.Services;

public interface IViewportPersistenceService
{
    /// <summary>
    /// Saves the viewport state for the given Anima to {configDirectory}/{animaId}.viewport.json.
    /// </summary>
    Task SaveAsync(string animaId, double scale, double panX, double panY, CancellationToken ct = default);

    /// <summary>
    /// Loads the viewport state for the given Anima. Returns null if no file exists.
    /// </summary>
    Task<ViewportState?> LoadAsync(string animaId, CancellationToken ct = default);
}

public sealed record ViewportState(double Scale, double PanX, double PanY);
```

Create `src/OpenAnima.Core/Services/ViewportPersistenceService.cs`:

```csharp
using System.Text.Json;
using Microsoft.Extensions.Logging;

namespace OpenAnima.Core.Services;

public class ViewportPersistenceService : IViewportPersistenceService
{
    private readonly string _configDirectory;
    private readonly ILogger<ViewportPersistenceService> _logger;

    private static readonly JsonSerializerOptions JsonOptions = new()
    {
        WriteIndented = false,
        PropertyNamingPolicy = JsonNamingPolicy.CamelCase
    };

    public ViewportPersistenceService(string configDirectory, ILogger<ViewportPersistenceService> logger)
    {
        _configDirectory = configDirectory;
        _logger = logger;
    }

    public async Task SaveAsync(string animaId, double scale, double panX, double panY, CancellationToken ct = default)
    {
        if (!Directory.Exists(_configDirectory))
            Directory.CreateDirectory(_configDirectory);

        var state = new ViewportState(scale, panX, panY);
        var filePath = Path.Combine(_configDirectory, $"{animaId}.viewport.json");
        await using var stream = File.Create(filePath);
        await JsonSerializer.SerializeAsync(stream, state, JsonOptions, ct);
    }

    public async Task<ViewportState?> LoadAsync(string animaId, CancellationToken ct = default)
    {
        var filePath = Path.Combine(_configDirectory, $"{animaId}.viewport.json");
        if (!File.Exists(filePath))
            return null;

        try
        {
            await using var stream = File.OpenRead(filePath);
            var state = await JsonSerializer.DeserializeAsync<ViewportState>(stream, JsonOptions, ct);
            return state;
        }
        catch (Exception ex)
        {
            _logger.LogWarning(ex, "Failed to load viewport for Anima {AnimaId}, using defaults", animaId);
            return null;
        }
    }
}
```

   ```csharp
   private async void TriggerViewportSave()
   {
       _viewportSaveDebounce?.Cancel();
       _viewportSaveDebounce?.Dispose();
       _viewportSaveDebounce = new CancellationTokenSource();

       try
       {
           await Task.Delay(1000, _viewportSaveDebounce.Token);

           var activeId = _animaContext.ActiveAnimaId;
           if (activeId != null)
           {
               await _viewportPersistence.SaveAsync(activeId, Scale, PanX, PanY, _viewportSaveDebounce.Token);
               _logger.LogDebug("Auto-saved viewport for Anima {AnimaId}: scale={Scale}, pan=({PanX},{PanY})", activeId, Scale, PanX, PanY);
           }
       }
       catch (OperationCanceledException)
       {
       }
       catch (Exception ex)
       {
           _logger.LogError(ex, "Viewport auto-save failed");
       }
   }
   ```
