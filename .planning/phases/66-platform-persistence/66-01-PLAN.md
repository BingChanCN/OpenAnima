# Plan 66-01: ChatHistoryService and chat.db initialization

## Objective

Create the ChatPersistence layer: `ChatDbConnectionFactory`, `ChatDbInitializer` (creates the `chat_messages` table), and `ChatHistoryService` (read/write per-Anima chat messages). Register everything in DI. This plan establishes the storage infrastructure that Plan 66-03 will wire into ChatPanel.

## must_haves

- `chat.db` created at `data/chat.db` on startup with `chat_messages` table
- `IChatHistoryService.SaveMessageAsync(animaId, role, content, toolCallsJson, inputTokens, outputTokens)` persists a row
- `IChatHistoryService.GetMessagesAsync(animaId)` returns all messages for the given Anima in ascending `created_at` order
- `IChatHistoryService.MarkInterruptedAsync(animaId)` marks all rows where `is_streaming = 1` as `is_streaming = 0` with content suffixed by ` [interrupted]`
- `ChatDbConnectionFactory` uses `Busy Timeout=5000` in the production constructor (same as `RunDbConnectionFactory`)
- All services registered as singletons in DI

using Dapper;
using Microsoft.Extensions.Logging;

namespace OpenAnima.Core.ChatPersistence;

public class ChatDbInitializer
{
    private readonly ChatDbConnectionFactory _factory;
    private readonly ILogger<ChatDbInitializer>? _logger;

    private const string SchemaScript = """
        CREATE TABLE IF NOT EXISTS chat_messages (
            id              INTEGER PRIMARY KEY AUTOINCREMENT,
            anima_id        TEXT NOT NULL,
            role            TEXT NOT NULL,
            content         TEXT NOT NULL,
            tool_calls_json TEXT,
            input_tokens    INTEGER NOT NULL DEFAULT 0,
            output_tokens   INTEGER NOT NULL DEFAULT 0,
            is_streaming    INTEGER NOT NULL DEFAULT 0,
            created_at      TEXT NOT NULL
        );

        CREATE INDEX IF NOT EXISTS idx_chat_messages_anima ON chat_messages(anima_id, created_at);
        """;

    public ChatDbInitializer(ChatDbConnectionFactory factory, ILogger<ChatDbInitializer>? logger = null)
    {
        _factory = factory ?? throw new ArgumentNullException(nameof(factory));
        _logger = logger;
    }

    public async Task EnsureCreatedAsync()
    {
        await using var conn = _factory.CreateConnection();
        await conn.OpenAsync();

        await conn.ExecuteAsync("PRAGMA journal_mode=WAL;");
        await conn.ExecuteAsync("PRAGMA synchronous=NORMAL;");
        await conn.ExecuteAsync(SchemaScript);

        _logger?.LogInformation("ChatDb schema ready");
    }
}
```

namespace OpenAnima.Core.ChatPersistence;

public interface IChatHistoryService
{
    /// <summary>
    /// Persists a single message for the given Anima.
    /// toolCallsJson may be null if the message has no tool calls.
    /// isStreaming should be true while the assistant is still generating; call SaveMessageAsync
    /// again with isStreaming=false once the stream completes (or use MarkInterruptedAsync on restart).
    /// </summary>
    Task SaveMessageAsync(
        string animaId,
        string role,
        string content,
        string? toolCallsJson,
        int inputTokens,
        int outputTokens,
        bool isStreaming = false);

    /// <summary>
    /// Returns all messages for the given Anima, ordered by created_at ASC.
    /// </summary>
    Task<IReadOnlyList<ChatHistoryRow>> GetMessagesAsync(string animaId);

    /// <summary>
    /// Updates the last row's content with finalized content and marks is_streaming=0.
    /// Called when streaming completes.
    /// </summary>
    Task FinalizeLastMessageAsync(string animaId, string finalContent);

    /// <summary>
    /// Marks all is_streaming=1 rows for the given Anima as is_streaming=0
    /// and appends " [interrupted]" to their content.
    /// Called at startup to handle incomplete streams from a previous session.
    /// </summary>
    Task MarkInterruptedAsync(string animaId);

    /// <summary>
    /// Deletes all messages for the given Anima.
    /// </summary>
    Task ClearHistoryAsync(string animaId);
}

public sealed record ChatHistoryRow(
    long Id,
    string AnimaId,
    string Role,
    string Content,
    string? ToolCallsJson,
    int InputTokens,
    int OutputTokens,
    bool IsStreaming,
    string CreatedAt);
```

```csharp
using Dapper;
using Microsoft.Extensions.Logging;

namespace OpenAnima.Core.ChatPersistence;

public class ChatHistoryService : IChatHistoryService
{
    private readonly ChatDbConnectionFactory _factory;
    private readonly ILogger<ChatHistoryService> _logger;

    public ChatHistoryService(ChatDbConnectionFactory factory, ILogger<ChatHistoryService> logger)
    {
        _factory = factory ?? throw new ArgumentNullException(nameof(factory));
        _logger = logger;
    }

    public async Task SaveMessageAsync(
        string animaId,
        string role,
        string content,
        string? toolCallsJson,
        int inputTokens,
        int outputTokens,
        bool isStreaming = false)
    {
        await using var conn = _factory.CreateConnection();
        await conn.OpenAsync();
        await conn.ExecuteAsync(
            """
            INSERT INTO chat_messages
                (anima_id, role, content, tool_calls_json, input_tokens, output_tokens, is_streaming, created_at)
            VALUES
                (@AnimaId, @Role, @Content, @ToolCallsJson, @InputTokens, @OutputTokens, @IsStreaming, @CreatedAt)
            """,
            new
            {
                AnimaId = animaId,
                Role = role,
                Content = content,
                ToolCallsJson = toolCallsJson,
                InputTokens = inputTokens,
                OutputTokens = outputTokens,
                IsStreaming = isStreaming ? 1 : 0,
                CreatedAt = DateTimeOffset.UtcNow.ToString("O")
            });
    }

    public async Task<IReadOnlyList<ChatHistoryRow>> GetMessagesAsync(string animaId)
    {
        await using var conn = _factory.CreateConnection();
        await conn.OpenAsync();
        var rows = await conn.QueryAsync<ChatHistoryRow>(
            """
            SELECT
                id           AS Id,
                anima_id     AS AnimaId,
                role         AS Role,
                content      AS Content,
                tool_calls_json AS ToolCallsJson,
                input_tokens AS InputTokens,
                output_tokens AS OutputTokens,
                is_streaming  AS IsStreaming,
                created_at   AS CreatedAt
            FROM chat_messages
            WHERE anima_id = @AnimaId
            ORDER BY created_at ASC, id ASC
            """,
            new { AnimaId = animaId });
        return rows.ToList();
    }

    public async Task FinalizeLastMessageAsync(string animaId, string finalContent)
    {
        await using var conn = _factory.CreateConnection();
        await conn.OpenAsync();
        await conn.ExecuteAsync(
            """
            UPDATE chat_messages
            SET content = @Content, is_streaming = 0
            WHERE id = (
                SELECT id FROM chat_messages
                WHERE anima_id = @AnimaId AND is_streaming = 1
                ORDER BY id DESC LIMIT 1
            )
            """,
            new { AnimaId = animaId, Content = finalContent });
    }

    public async Task MarkInterruptedAsync(string animaId)
    {
        await using var conn = _factory.CreateConnection();
        await conn.OpenAsync();
        var affected = await conn.ExecuteAsync(
            """
            UPDATE chat_messages
            SET content = content || ' [interrupted]', is_streaming = 0
            WHERE anima_id = @AnimaId AND is_streaming = 1
            """,
            new { AnimaId = animaId });
        if (affected > 0)
            _logger.LogInformation("Marked {Count} interrupted chat messages for Anima {AnimaId}", affected, animaId);
    }

    public async Task ClearHistoryAsync(string animaId)
    {
        await using var conn = _factory.CreateConnection();
        await conn.OpenAsync();
        await conn.ExecuteAsync(
            "DELETE FROM chat_messages WHERE anima_id = @AnimaId",
            new { AnimaId = animaId });
    }
}
```

public static IServiceCollection AddChatPersistenceServices(
    this IServiceCollection services,
    string? dataRoot = null)
{
    dataRoot ??= Path.Combine(AppContext.BaseDirectory, "data");
    var dbPath = Path.Combine(dataRoot, "chat.db");

    Directory.CreateDirectory(dataRoot);

}
